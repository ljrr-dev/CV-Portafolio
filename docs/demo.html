import pandas as pd
import plotly.express as px
import plotly.graph_objects as go

# Reemplaza esto con tus datos reales
# Ejemplo: datos de indicadores ODS de World Bank vía wbgapi,
# o cualquier CSV que quieras dejar precargado en el repo.

df = pd.DataFrame({
    "año": list(range(2015, 2024)),
    "valor_a": [10, 15, 13, 18, 22, 25, 21, 28, 30],
    "valor_b": [8,  12, 14, 17, 19, 23, 24, 26, 29],
})

Tendencia temporal

Code 
fig = px.line(
    df,
    x="año",
    y=["valor_a", "valor_b"],
    labels={"value": "Indicador", "año": "Año", "variable": "Serie"},
    title="Evolución de indicadores — reemplaza con tus datos",
    template="plotly_dark",
)
fig.update_layout(
    font_family="'IBM Plex Mono', monospace",
    plot_bgcolor="rgba(0,0,0,0)",
    paper_bgcolor="rgba(0,0,0,0)",
)
fig.show()
